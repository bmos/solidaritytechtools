1"""Assign average daily traffic (AADT) to coordinates for yard-sign prioritization. 

2 

3WisDOT publishes ~27k statewide traffic-count points, each with a RDWY_AADT value 

4(Annual Average Daily Traffic = average vehicles/day on that road segment) and a 

5lat/lon. Given a coordinate, we snap it to the nearest count point and report that 

6point's AADT as the location's "traffic score". If the nearest count point is 

7farther than ``max_distance_m``, the score is "no data" (aadt=None) rather than 

8attributing a distant road's traffic to the address. 

9 

10Use this to build Solidarity Tech pull lists that prioritize people with the most 

11cars passing their house: score each member's coordinates, then sort high-to-low. 

12 

13This module is self-contained (stdlib + httpx). If ``TRAFFIC_GEOJSON_PATH`` is 

14None, the WisDOT GeoJSON is downloaded to a cached file in the system temp dir on 

15first use. 

16"""

178def _normalize_road_name(text: str) -> str: 

179 """Reduce a road name or count description to a bare, comparable street name. 

180 

181 Drops route designations, directionals, street-type suffixes, and bare house/route 

182 numbers. Returns "" when nothing nameable remains (e.g. an unnamed highway segment). 

183 """ 

184 upper = text.upper() 

185 upper = _ROUTE_RE.sub(" ", upper) 

186 upper = re.sub(r"[^A-Z0-9 ]", " ", upper) 

187 tokens = [ 

188 tok 

189 for tok in upper.split() 

190 if tok not in _DIRECTIONALS 

191 and tok not in _STREET_TYPES 

192 and tok not in _ROUTE_CLASSES 

193 and not tok.isdigit() # bare house/route numbers; keeps ordinals like "76TH" 

194 ] 

195 return " ".join(tokens)

297def _haversine_m(lat1: float, lon1: float, lat2: float, lon2: float) -> float: 

298 """Great-circle distance in meters between two lat/lon points.""" 

299 p1, p2 = math.radians(lat1), math.radians(lat2) 

300 dphi = math.radians(lat2 - lat1) 

301 dlmb = math.radians(lon2 - lon1) 

302 a = math.sin(dphi / 2) ** 2 + math.cos(p1) * math.cos(p2) * math.sin(dlmb / 2) ** 2 

303 return 2 * _EARTH_RADIUS_M * math.asin(math.sqrt(a))

306def _download_geojson(url: str = TRAFFIC_GEOJSON_URL) -> Path: 

307 """Download the WisDOT GeoJSON to a cached temp file, reusing it if present.""" 

308 cache_path = Path(tempfile.gettempdir()) / TEMP_GEOJSON_FILENAME 

309 if cache_path.exists() and cache_path.stat().st_size > 0: 

310 return cache_path 

311 tmp_path = cache_path.with_suffix(cache_path.suffix + ".part") 

312 with httpx.stream("GET", url, follow_redirects=True, timeout=_DOWNLOAD_TIMEOUT_S) as response: 

313 response.raise_for_status() 

314 with open(tmp_path, "wb") as f: 

315 for chunk in response.iter_bytes(): 

316 f.write(chunk) 

317 tmp_path.replace(cache_path) 

318 return cache_path

333 for feature in data.get("features", []): 

334 props = feature.get("properties") or {} 

335 geometry = feature.get("geometry") or {} 

336 coords = geometry.get("coordinates") 

337 aadt = props.get(_AADT_FIELD)

339 continue 

340 lon, lat = coords[0], coords[1]

343 name = _normalize_road_name(primary) 

344 for route_key, route_name in _extract_route_names(location): 344 ↛ 345line 344 didn't jump to line 345 because the loop on line 344 never started

345 route_named.setdefault(route_key, []).append((lat, lon, route_name)) 

346 raw.append(

348 "lat": lat, 

349 "lon": lon,

380class TrafficScorer: 

381 """Builds a spatial index over the count points for fast nearest lookups. 

382 

383 Loading and indexing the ~27k points is done once at construction; scoring is 

384 then cheap, so reuse a single instance across many addresses. 

385 """ 

386 

387 def __init__(self, geojson_path: Path | None = None) -> None: 

388 path = geojson_path if geojson_path is not None else _resolve_geojson_path() 

389 self.points: list[CountPoint] = _load_points(path) 

390 self._grid: dict[tuple[int, int], list[CountPoint]] = {} 

391 for point in self.points: 

392 self._grid.setdefault(self._cell(point.lat, point.lon), []).append(point) 

393 

394 def _cell(self, lat: float, lon: float) -> tuple[int, int]: 

395 return (int(lat / _GRID_CELL_DEG), int(lon / _GRID_CELL_DEG)) 

396 

397 def _nearest( 

398 self, 

399 lat: float, 

400 lon: float, 

401 stop_distance_m: float, 

402 road_name: str | None = None, 

403 exclude_freeways: bool = True, 

404 ) -> tuple[CountPoint | None, float]: 

405 """Return (nearest_point, distance_m), searching outward in grid rings. 

406 

407 Abandons once it's certain the nearest point is farther than 

408 ``stop_distance_m`` (a definitive "no data"); the returned distance is then 

409 a lower bound, which keeps no-data lookups in empty areas from scanning far. 

410 

411 When ``road_name`` is given, only count points on that same named road are 

412 considered, so an address never inherits a parallel street's traffic. When 

413 ``exclude_freeways`` is set, interstate/ramp/expressway counts are skipped. 

414 """ 

415 ci, cj = self._cell(lat, lon) 

416 best: CountPoint | None = None 

417 best_d = math.inf 

418 # Smallest cell edge in meters (longitude shrinks with latitude); once a 

419 # candidate is closer than the radius fully covered by searched rings, 

420 # nothing unsearched can be nearer. 

421 cell_m = _GRID_CELL_DEG * 111_320 * max(math.cos(math.radians(lat)), 0.1) 

422 ring = 0 

423 while True: 

424 for i in range(ci - ring, ci + ring + 1): 

425 for j in range(cj - ring, cj + ring + 1): 

426 # Only the outer shell of the (2*ring+1) box is new each ring. 

427 if ring > 0 and ci - ring < i < ci + ring and cj - ring < j < cj + ring: 

428 continue 

429 for point in self._grid.get((i, j), ()): 

430 if road_name is not None and point.road_name != road_name: 

431 continue 

432 if exclude_freeways and point.is_freeway: 

433 continue 

434 d = _haversine_m(lat, lon, point.lat, point.lon) 

435 if d < best_d: 

436 best_d, best = d, point 

437 covered = ring * cell_m 

438 if best is not None and best_d <= covered: 

439 break 

440 if covered >= stop_distance_m and best_d > stop_distance_m: 

441 break 

442 ring += 1 

443 if ring * cell_m > _MAX_SEARCH_M: 

444 break 

445 return best, best_d

447 def score( 

448 self, 

449 lat: float, 

450 lon: float, 

451 *, 

452 street_hint: str | None = None, 

453 max_distance_m: float = DEFAULT_MAX_DISTANCE_M, 

454 street_match_max_distance_m: float = DEFAULT_STREET_MATCH_MAX_DISTANCE_M, 

455 fallback_to_nearest: bool = False, 

456 exclude_freeways: bool = True, 

457 ) -> TrafficScore: 

458 """Score a coordinate, optionally constrained to a known street. 

459 

460 Without ``street_hint``, snaps to the nearest count within ``max_distance_m``. 

461 

462 With ``street_hint`` (the member's street line, e.g. "2600 N Oakland Ave"), 

463 snaps only to a count *on that same road* within ``street_match_max_distance_m``. 

464 This is far more accurate in a dense grid: a quiet street no longer inherits a 

465 parallel arterial's count. If there's no count on that road nearby, the result 

466 is "no data" — unless ``fallback_to_nearest`` is set, which then falls back to 

467 the plain nearest-count snap within ``max_distance_m``. 

468 

469 ``exclude_freeways`` (default True) drops interstate/ramp/expressway counts so a 

470 house never inherits a freeway's volume — set False only if you actually want it. 

471 """ 

472 hint_name = _hint_road_name(street_hint) if street_hint else "" 

473 if hint_name: 

474 point, dist = self._nearest( 

475 lat, 

476 lon, 

477 stop_distance_m=street_match_max_distance_m, 

478 road_name=hint_name, 

479 exclude_freeways=exclude_freeways, 

480 ) 

481 if point is not None and dist <= street_match_max_distance_m: 

482 return _to_score(point, dist) 

483 if not fallback_to_nearest: 

484 return _NO_DATA 

485 

486 point, dist = self._nearest( 

487 lat, lon, stop_distance_m=max_distance_m, exclude_freeways=exclude_freeways 

488 ) 

489 if point is None or dist > max_distance_m: 

490 return _NO_DATA 

491 return _to_score(point, dist)

494 self,
